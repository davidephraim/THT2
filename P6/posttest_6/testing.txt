<script>	
    let siswa={
        nama:"David",
        umur:10,
        nilai:[85,90,78,92,88],
        alamat:{
            jalan:"JL Cipulir VI",
            kota:"Jakarta",
        },
        lulus: function(){
            let totalNilai=0
            for(let i=0; i<this.nilai.length; ++i) totalNilai += this.nilai[i];
            
            let rataRata = totalNilai / this.nilai.length;

            if(rataRata >= 70) return true;
            else return false
        },
    }

    let rataRata = siswa.nilai

    document.getElementById("nama").innerText = siswa.nama;
    document.getElementById("umur").textContent = siswa.umur;
    document.getElementById("alamat").textContent = siswa.alamat.kota + ", " + siswa.alamat.jalan;
    document.getElementById("nilai").textContent = siswa.nilai.join(", ");
    document.getElementById("rata-rata").textContent = rataRata;

    if(siswa.lulus()) document.getElementById("rata-rata").textContent = "Siswa telah lulus";
    else document.getElementById("rata-rata").textContent = "Siswa tidak lulus";
